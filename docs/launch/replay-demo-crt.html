export async function agent(input, chidori) {
  const plan = await chidori.prompt("Plan: " + input.bug);
  const diff = await chidori.tool("repo.apply", { plan });
  await chidori.input("Ship it?", { type: "approval" });
  return chidori.prompt("Changelog:\n" + diff);
}
